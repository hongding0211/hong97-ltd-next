export default BlogContainer

export async function getServerSideProps(context) {
  const { locale, query } = context
  const { key } = query || {}
  const title = BlogConfig.find(item => item.key === key)?.title || ''
  return {
    props: {
      ...(await serverSideTranslations(locale, ['common'])),
      title,
    },
  }
}
