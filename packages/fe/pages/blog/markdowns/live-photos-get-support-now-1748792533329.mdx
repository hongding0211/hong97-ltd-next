export async function getServerSideProps(context) {
  const { locale, query } = context
  const meta = await http.get('GetBlogMeta', {
    blogId: query?.key
  })
  time.setLocale(locale)
  return {
    props: {
      ...(await serverSideTranslations(locale, ['common', 'blog'])),
      meta: meta?.data,
      locale,
    },
  }
}
